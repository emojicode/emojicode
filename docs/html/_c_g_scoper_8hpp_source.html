51  void pushScope() {
52  scopes_.emplace_back(scopes_.empty() ? 0 : scopes_.back().maxIndex);

59  void popScope(InstructionCount count) {
60  auto &scope = scopes_.back();
61  reduceOffsetBy(scope.size);
62  for (auto it = variables_.begin() ; it < variables_.begin(); it++) {
63  if (it->initialized == 1) {
64  it->type.objectVariableRecords(it->stackIndex.value(), &fovInfo_, it->initPosition, count);
65  }
